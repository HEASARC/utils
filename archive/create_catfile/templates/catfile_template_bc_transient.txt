# Variables within settings should be named in upper case with ':' around
# them. Some canned settings are already available: new ones can be 
# defined (or redefined) with Unix (NOT regex) style wildcards
# E.g. variable HEALPIX [0-9][0-9][0-9] for three-digit Healpixel number
# PATH (or ROOTPATH), SUBDIR, and CATNAME must be defined
#

# These variables describe where to find the data (path/subdir/)
# Transient/trigger data OBS_ID is in the form YYMMDDXXX, with 
# first possible year being 2022 and latest is 2029
variable PATH /processing/burstcube/fake_data/final/transient/
variable OBS_ID 2[2-9][01][0-9][0-3][0-9][0-9][0-9][0-9]
variable SUBDIR 202[2-9]_[01][0-9]/:OBS_ID:/

# These variables define the name and the destination of the catalog file
variable CATNAME bc:OBS_ID:.cat
variable CATDEST products
