# Generic FITS catalog (.cat) file generator template settings
# for make_generic_catfile.pl (v 0.4 March 2, 2020)
# 
# XLCalibur data template
# Created 04 August, 2025
#
# Variables within settings should be named in upper case with ':' around
# them. Some canned settings are already available: new ones can be 
# defined (or redefined) with Unix (NOT regex) style wildcards
# E.g. variable HEALPIX [0-9][0-9][0-9] for three-digit Healpixel number
# PATH (or ROOTPATH), SUBDIR, and CATNAME must be defined
#  

# These variables describe where to find the data (path/subdir/)
# OBS_ID is in the form YYYMMDD, with first possible value being
# 20151006

variable PATH /FTP/.calet_archive/cgbm_v3.2a_240523/obs
variable OBS_ID 20[12][0-9][01][0-9][0-3][0-9]
variable TRIG_ID [0-2][0-9][0-5][0-9][0-5][0-9]
variable SUBDIR 20[12][0-9]/:OBS_ID:/

# These variables define the name and the destination of the catalog file
variable CATNAME cgbm_:OBS_ID:.cat
variable CATDEST auxil

# Additional mission specific variables to find the files

filelist auxil/cgbm_:OBS_ID:*
filelist monitor/cgbm_:OBS_ID:*
filelist events/cgbm_:OBS_ID:*
filelist products/cgbm_:OBS_ID:*
