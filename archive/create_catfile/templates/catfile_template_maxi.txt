# Generic FITS catalog (.cat) file generator template settings
# v 0.2 Dec 8, 2019
# 
# MAXI event and auxil data catalog summary file
#
#
# Variables within settings should be named in upper case with ':' around
# them. Some canned settings are already available: new ones can be 
# defined (or redefined) with Unix (NOT regex) style wildcards
# E.g. variable HEALPIX [0-9][0-9][0-9] for three-digit Healpixel number
# PATH (or ROOTPATH) and CATNAME must be defined
#
# Write to _v23 branch only
#variable PATH /FTP/.maxi_archive/
variable PATH /nasA_maxi1/ftp/.ver2.3/mxdata/obs/
variable SUBDIR MJD:MJD1000:_mod/MJD:MJD:/
variable HEALPIX [0-9][0-9][0-9]
variable CATNAME mx_mjd:MJD:.cat
#
# Range limit
# Constrain search to only a subset of possible values for variables above
# Can be overrided on command line
#
range MJD 56700 56709
#
# Template for sets of files 
#
filelist auxil/mx_mjd:MJD:*
filelist events/gsc_low/mx_mjd:MJD:_gsc_low_:HEALPIX:.evt.gz
filelist events/gsc_med/mx_mjd:MJD:_gsc_med_:HEALPIX:.evt.gz
filelist events/ssc_med/mx_mjd:MJD:_ssch_med_:HEALPIX:.evt.gz
filelist events/ssc_med/mx_mjd:MJD:_sscz_med_:HEALPIX:.evt.gz
